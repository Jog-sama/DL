import numpy as np

        def sigmoid(x):
             return 1 / (1 + np.exp(-x))


        # test predictions
        dataset = np.array([
            [2.3, 4.5, -1],
            [1.2, 3.8, -1],
            [5.0, 2.0, -1],
            [3.5, 1.0, -1],
            [2.0, 2.0, -1],
        ])

        weights1 = np.array([
        [2, 3, -1],
        [-5, 7, 2],
        [1, -4, 6],
        [-8, 2, 3]
        ])


        transposed_dataset = dataset.T
        transposed_dataset

        hidden_layer_1_input = np.dot(weights1, transposed_dataset)
        hidden_layer_1_output = sigmoid(hidden_layer_1_input)
        hidden_layer_1_output

        weights2 = np.array([
        [1, -1, 3, 2]
         ])


        final_output = sigmoid(np.dot(weights2, hidden_layer_1_output))
